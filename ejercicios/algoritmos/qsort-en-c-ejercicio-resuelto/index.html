#include <stdio.h>
#include <stdlib.h>

int cmp_asc(const void *a, const void *b) {
    int x = *(const int *)a;
    int y = *(const int *)b;
    return (x > y) - (x < y);
}

int main(void) {
    int v[] = {9, 1, 5, 3, 7};
    int n = (int)(sizeof(v) / sizeof(v[0]));

    qsort(v, n, sizeof(int), cmp_asc);
    for (int i = 0; i < n; i++) printf("%d ", v[i]);
    printf("\n");
    return 0;
}
