#include <ctype.h>
#include <stdio.h>

int contar_palabras(const char *s) {
    int total = 0;
    int en_palabra = 0;

    for (int i = 0; s[i] != '\0'; i++) {
        if (!isspace((unsigned char)s[i]) && !en_palabra) {
            total++;
            en_palabra = 1;
        } else if (isspace((unsigned char)s[i])) {
            en_palabra = 0;
        }
    }
    return total;
}

int main(void) {
    printf("%d\n", contar_palabras("C   en rigle dev"));
    return 0;
}
