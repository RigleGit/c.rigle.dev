#include <stdio.h>

#define N 4
int g[N][N] = {0};

void add_edge(int u, int v) { g[u][v] = 1; g[v][u] = 1; }

int main(void) {
    add_edge(0, 1); add_edge(0, 2);
    printf("0 conectado con 1: %d\n", g[0][1]);
    printf("2 conectado con 3: %d\n", g[2][3]);
    return 0;
}
