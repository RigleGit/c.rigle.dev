Problem statement

Define a Student struct with name and grade. Store multiple students in an array and compute average grade.

#include <stdio.h>

typedef struct {
    char name[20];
    float grade;
} Student;

int main(void) {
    Student s[] = {
        {"Ana", 8.0f},
        {"Luis", 7.5f},
        {"Marta", 9.0f}
    };

    float sum = 0.0f;
    int n = sizeof(s) / sizeof(s[0]);

    for (int i = 0; i < n; i++) sum += s[i].grade;
    printf("Average: %.2f\n", sum / n);

    return 0;
}
